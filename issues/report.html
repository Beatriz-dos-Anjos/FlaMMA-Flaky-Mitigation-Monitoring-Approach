import json import argparse from datetime import datetime def load_report(path): """Carrega o JSON do relatório de testes.""" with open(path, "r") as f: return json.load(f) def generate_html(report_path="flaky_report.json", out_path="report.html"): """Gera o relatório HTML baseado no JSON.""" data = load_report(report_path) # Cabeçalho do HTML timestamp = datetime.now().strftime("%Y-%m-%d %H:%M:%S") html = """ Relatório de Testes Flaky

Relatório de Testes Flaky

Gerado em: __REPORT_TIMESTAMP__

""" # Loop para preencher a tabela com os testes for test_name, info in data["tests"].items(): passes = info["pass_count"] fails = info["fail_count"] total = passes + fails fail_rate = (fails / total) * 100 if total > 0 else 0 row_class = "flaky" if info["is_flaky"] else "stable" html += f""" """ # Fechamento do HTML html += """
Teste Passes Fails Runs Porcentagem Falhas Status
{test_name} {passes} {fails} {info["runs"]} {fail_rate:.1f}% {"Flaky" if info["is_flaky"] else "Estável"}
""" # Insere timestamp real e grava o arquivo HTML html = html.replace("__REPORT_TIMESTAMP__", timestamp) # Grava o arquivo HTML with open(out_path, "w") as f: f.write(html) f.write(html) print(f"Relatório gerado em: {out_path}") def main(): parser = argparse.ArgumentParser() parser.add_argument("--report", default="flaky_report.json", help="Caminho do JSON de entrada") parser.add_argument("--out", default="report.html", help="Caminho do HTML de saída") args = parser.parse_args() generate_html(args.report, args.out) if __name__ == "__main__": main()
