import json from datetime import datetime import argparse def generate_html(report_path=".flamma/quarantine/flaky_tests.json", out_path="report.html", workflow_name=None): with open(report_path, "r") as f: data = json.load(f) workflow_name = workflow_name or "Último workflow" html = f""" Flaky Test Report

Relatório de Testes Flaky

Workflow: {workflow_name}

Gerado em: {datetime.now().strftime("%Y-%m-%d %H:%M:%S")}

""" for test_name, info in data["tests"].items(): passes = info["pass_count"] fails = info["fail_count"] total = passes + fails fail_rate = (fails / total) * 100 if total > 0 else 0 row_class = "flaky" if info["is_flaky"] else "stable" html += f""" """ html += """
Teste Passes Fails Runs Porcentagem Falhas Status
{test_name} {passes} {fails} {info["runs"]} {fail_rate:.1f}% {"Flaky" if info["is_flaky"] else "Estável"}
""" with open(out_path, "w") as f: f.write(html) print(f"Relatório gerado em: {out_path}") def main(): parser = argparse.ArgumentParser() parser.add_argument("--report", default=".flamma/quarantine/flaky_tests.json") parser.add_argument("--out", default="report.html") parser.add_argument("--workflow", help="Nome do workflow ou commit") args = parser.parse_args() generate_html(args.report, args.out, args.workflow) if __name__ == "__main__": main()
